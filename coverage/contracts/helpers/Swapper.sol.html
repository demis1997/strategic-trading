pragma solidity ^0.8.24;
 
import { ISwapper } from "../interfaces/ISwapper.sol";
import { AccessControlUpgradeable } from "@openzeppelin/contracts-upgradeable/access/AccessControlUpgradeable.sol";
import { SafeERC20 } from "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";
import { IERC20 } from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
 
contract Swapper is AccessControlUpgradeable, ISwapper {
    using SafeERC20 for IERC20;
 
    /// @dev Roles for executing swaps and resolving swap data
    bytes32 public constant SWAPPER_EXECUTOR = keccak256("SWAPPER_EXECUTOR");
    bytes32 public constant SWAPPER_RESOLVER = keccak256("SWAPPER_RESOLVER");
 
    /// @notice Default slippage percentage used in swaps (in basis points)
    uint256 public defaultSlippage;
 
    /// @notice Struct to store DEX data including location and swap function
    struct DexData {
        address dexLocation;
        bytes swapFunction;
    }
 
    /// @notice Mapping to store DexData for specific token pairs
    mapping(address => mapping(address => DexData)) public swapData;
 
    /// @notice Initializes the Swapper contract with default slippage and roles
    /// @param _defaultSlippage Initial default slippage percentage (e.g., 4% as 400 basis points)
    /// @param adminAddress Address to be granted the DEFAULT_ADMIN_ROLE and other roles
    function initialize(uint256 _defaultSlippage, address adminAddress) public initializer {
        if (_defaultSlippage <= 0) revert InvalidSlippage(_defaultSlippage);
        if (adminAddress == address(0)) revert InvalidAddress(adminAddress);
 
        defaultSlippage = _defaultSlippage;
        _grantRole(DEFAULT_ADMIN_ROLE, adminAddress);
        _grantRole(SWAPPER_EXECUTOR, adminAddress);
        _grantRole(SWAPPER_RESOLVER, adminAddress);
    }
 
    /// @notice Sets the default slippage for swaps
    /// @param slippage The new default slippage percentage (in basis points)
    function setDefaultSlippage(uint256 slippage) external override onlyRole(DEFAULT_ADMIN_ROLE) {
        emit DefaultSlippageUpdated(defaultSlippage, slippage);
        defaultSlippage = slippage;
    }
 
    /// @notice Updates the DexData for a specific token pair
    /// @param tokenIn The input token address
    /// @param tokenOut The output token address
    /// @param dexData The new DexData for the token pair
    function setDexData(
        address tokenIn,
        address tokenOut,
        DexData calldata dexData
    ) external onlyRole(SWAPPER_RESOLVER) {
        if (tokenIn == address(0) || tokenOut == address(0)) revert ZeroAddressNotAllowed();
        swapData[tokenIn][tokenOut] = dexData;
    }
 
    /// @notice Swaps a specified amount of one token for another
    /// @param owner The address of the token owner
    /// @param receiver The address receiving the swapped tokens
    /// @param amount The amount of tokenIn to swap
    /// @param tokenIn The address of the token being swapped
    /// @param tokenOut The address of the token to receive
    /// @param slippage The acceptable slippage percentage for this swap
    /// @return swpAmount The amount of tokenOut received
    function swap(
        address owner,
        address receiver,
        uint256 amount,
        address tokenIn,
        address tokenOut,
        uint256 slippage,
        uint256 flags
    ) external onlyRole(SWAPPER_EXECUTOR) returns (uint256 swpAmount) {
        DexData memory dexData = swapData[tokenIn][tokenOut];
        if (dexData.dexLocation == address(0)) revert TokenNotAllowed(tokenIn, tokenOut);
 
        IERC20(tokenIn).safeTransferFrom(owner, address(this), amount);
 
        (bool success, bytes memory returnData) = dexData.dexLocation.call(dexData.swapFunction);
        require(success, "Swap execution failed");
 
        swpAmount = abi.decode(returnData, (uint256));
 
        uint256 minReturn = (amount * (10000 - slippage)) / 10000;
        require(swpAmount >= minReturn, "Received amount less than minimum slippage tolerance");
 
        IERC20(tokenOut).safeTransfer(receiver, swpAmount);
 
        emit SwapExecuted(owner, receiver, tokenIn, tokenOut, amount, swpAmount);
    }
}
